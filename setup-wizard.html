Google Sheets Sync
Saves student data to Google Sheets so it works on any Chromebook. Without this, data is lost if students clear their browser or switch devices.
⏱ 10–15 minutes (one-time only)
✓ Done
Step 1: Go to sheets.new and create a new Google Sheet. Name it CompTIA A+ Class Data.
Step 2: In the sheet, go to Extensions → Apps Script. Delete any existing code. Paste the script below and click Save (💾).
Step 3: Click Deploy → New Deployment. Type: Web App. Execute as: Me. Access: Anyone. Click Deploy.
Step 4: Google asks you to authorize. Click Advanced → Go to CompTIA A+ (unsafe) → Allow. This is normal for Apps Scripts.
Step 5: Copy the deployment URL (looks like https://script.google.com/macros/s/ABC.../exec). Paste it below.
// CompTIA A+ Platform — Google Apps Script Backend // Paste this entire script into Extensions > Apps Script function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { const p = e.parameter; const ss = SpreadsheetApp.getActiveSpreadsheet(); try { let result = {}; if (p.action === 'ping') { result = { ok: true, message: 'Connected!' }; } else if (p.action === 'save_score') { getOrCreate(ss,'Scores').appendRow([new Date(),p.code,p.day,p.type,p.score,p.pct]); result = { ok: true }; } else if (p.action === 'get_scores') { const data = getOrCreate(ss,'Scores').getDataRange().getValues(); result = { ok: true, scores: data.filter(r=>r[1]===p.code).slice(1) }; } else if (p.action === 'save_pbis') { getOrCreate(ss,'PBIS').appendRow([new Date(),p.code,p.reason,Number(p.pts),p.auto]); result = { ok: true }; } else if (p.action === 'get_pbis') { const data = getOrCreate(ss,'PBIS').getDataRange().getValues().slice(1); const by = {}; data.forEach(r => { by[r[1]] = (by[r[1]]||0) + Number(r[3]); }); result = { ok: true, totals: by }; } else if (p.action === 'save_note') { const s = getOrCreate(ss,'Notes'); const data = s.getDataRange().getValues(); const i = data.findIndex(r=>r[1]===p.code&&r[2]===p.chapter); const row = [new Date(),p.code,p.chapter,p.notes]; if (i > 0) s.getRange(i+1,1,1,4).setValues([row]); else s.appendRow(row); result = { ok: true }; } else if (p.action === 'save_assessment') { getOrCreate(ss,'Assessments').appendRow([new Date(),p.code,p.day,p.confidence,p.confusing,p.clicked]); result = { ok: true }; } else if (p.action === 'get_class_summary') { const scores = getOrCreate(ss,'Scores').getDataRange().getValues().slice(1); const pbis = getOrCreate(ss,'PBIS').getDataRange().getValues().slice(1); const codes = [...new Set(scores.map(r=>r[1]).filter(Boolean))]; const summary = codes.map(code => { const s = scores.filter(r=>r[1]===code); const t = s.filter(r=>r[3]==='timed').map(r=>Number(r[5])); const avg = t.length ? Math.round(t.reduce((a,b)=>a+b,0)/t.length) : 0; const pts = pbis.filter(r=>r[1]===code).reduce((s,r)=>s+Number(r[3]),0); return { code, avg, pts, sessions: s.length }; }); result = { ok: true, summary }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ok:false,error:err.message})) .setMimeType(ContentService.MimeType.JSON); } } function getOrCreate(ss, name) { let s = ss.getSheetByName(name); if (!s) { s = ss.insertSheet(name); const h = { Scores: ['Timestamp','Code','Day','Type','Score','Pct'], Notes: ['Timestamp','Code','Chapter','Notes'], PBIS: ['Timestamp','Code','Reason','Points','Auto'], Assessments: ['Timestamp','Code','Day','Confidence','Confusing','Clicked'], }; if (h[name]) { s.getRange(1,1,1,h[name].length).setValues([h[name]]).setFontWeight('bold'); s.setFrozenRows(1); } } return s; }
