⚙️ One-Time Setup

15 minutes · Free · No server needed
1
Create a Google Sheet
Open Google Sheets and create a new spreadsheet. Name it "CompTIA A+ Class Data". This is your database — it stores all student scores, notes, PBIS points, and question bank entries. Keep it in your Google Drive.
Open Google Sheets →
2
Open Apps Script
In your new Google Sheet, go to Extensions → Apps Script. Delete any existing code in the editor. Paste the entire script below, then click Save (the floppy disk icon).
// CompTIA A+ Classroom Platform — Google Apps Script Backend // Paste this entire script into Extensions > Apps Script const SHEET_ID = SpreadsheetApp.getActiveSpreadsheet().getId(); function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { const params = e.parameter; const action = params.action; const ss = SpreadsheetApp.getActiveSpreadsheet(); try { let result = {}; if (action === 'ping') { result = { ok: true, message: 'Connected!', sheets: ss.getSheets().map(s => s.getName()) }; } else if (action === 'save_score') { const sheet = getOrCreateSheet(ss, 'Scores'); sheet.appendRow([ new Date().toISOString(), params.code, params.day, params.type, params.score, params.pct, params.correct, params.total ]); result = { ok: true }; } else if (action === 'get_scores') { const sheet = getOrCreateSheet(ss, 'Scores'); const data = sheet.getDataRange().getValues(); const scores = data.slice(1).filter(r => r[1] === params.code).map(r => ({ ts: r[0], code: r[1], day: r[2], type: r[3], score: r[4], pct: r[5], correct: r[6], total: r[7] })); result = { ok: true, scores }; } else if (action === 'save_note') { const sheet = getOrCreateSheet(ss, 'Notes'); const data = sheet.getDataRange().getValues(); const existing = data.findIndex(r => r[1] === params.code && r[2] === params.chapter); const row = [new Date().toISOString(), params.code, params.chapter, params.notes]; if (existing > 0) { sheet.getRange(existing + 1, 1, 1, 4).setValues([row]); } else { sheet.appendRow(row); } result = { ok: true }; } else if (action === 'get_notes') { const sheet = getOrCreateSheet(ss, 'Notes'); const data = sheet.getDataRange().getValues(); const notes = data.slice(1).filter(r => r[1] === params.code).map(r => ({ chapter: r[2], notes: r[3], ts: r[0] })); result = { ok: true, notes }; } else if (action === 'save_pbis') { const sheet = getOrCreateSheet(ss, 'PBIS'); sheet.appendRow([ new Date().toISOString(), params.code, params.reason, params.pts, params.auto === 'true' ? 'Auto' : 'Manual' ]); result = { ok: true }; } else if (action === 'get_pbis') { const sheet = getOrCreateSheet(ss, 'PBIS'); const data = sheet.getDataRange().getValues(); const byCode = {}; data.slice(1).forEach(r => { if (!byCode[r[1]]) byCode[r[1]] = { total: 0, log: [] }; byCode[r[1]].total += Number(r[3]); byCode[r[1]].log.push({ ts: r[0], reason: r[2], pts: r[3], type: r[4] }); }); result = { ok: true, pbis: byCode }; } else if (action === 'save_question') { const sheet = getOrCreateSheet(ss, 'QuestionBank'); sheet.appendRow([ new Date().toISOString(), params.code, params.day, params.raw, params.cleaned, params.answer, params.topic ]); result = { ok: true }; } else if (action === 'get_questions') { const sheet = getOrCreateSheet(ss, 'QuestionBank'); const data = sheet.getDataRange().getValues(); const questions = data.slice(1).map(r => ({ ts: r[0], code: r[1], day: r[2], raw: r[3], cleaned: r[4], answer: r[5], topic: r[6] })); result = { ok: true, questions }; } else if (action === 'save_assessment') { const sheet = getOrCreateSheet(ss, 'SelfAssessments'); sheet.appendRow([ new Date().toISOString(), params.code, params.day, params.confidence, params.confusing, params.clicked ]); result = { ok: true }; } else if (action === 'get_class_summary') { const scores = getOrCreateSheet(ss, 'Scores').getDataRange().getValues().slice(1); const pbis = getOrCreateSheet(ss, 'PBIS').getDataRange().getValues().slice(1); const questions = getOrCreateSheet(ss, 'QuestionBank').getDataRange().getValues().slice(1); const students = [...new Set(scores.map(r => r[1]).filter(Boolean))]; const summary = students.map(code => { const studentScores = scores.filter(r => r[1] === code); const timedScores = studentScores.filter(r => r[3] === 'timed').map(r => Number(r[5])); const avg = timedScores.length ? Math.round(timedScores.reduce((a,b)=>a+b,0)/timedScores.length) : null; const pts = pbis.filter(r => r[1] === code).reduce((s,r) => s + Number(r[3]), 0); return { code, avg, pts, days: studentScores.length }; }); result = { ok: true, students: summary, totalQuestions: questions.length }; } return ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ ok: false, error: err.message })) .setMimeType(ContentService.MimeType.JSON); } } function getOrCreateSheet(ss, name) { let sheet = ss.getSheetByName(name); if (!sheet) { sheet = ss.insertSheet(name); const headers = { 'Scores': ['Timestamp','Student Code','Day','Type','Score','Pct','Correct','Total'], 'Notes': ['Timestamp','Student Code','Chapter','Notes JSON'], 'PBIS': ['Timestamp','Student Code','Reason','Points','Type'], 'QuestionBank': ['Timestamp','Student Code','Day','Raw Question','Cleaned Question','Answer','Topic'], 'SelfAssessments': ['Timestamp','Student Code','Day','Confidence','Most Confusing','What Clicked'] }; if (headers[name]) { sheet.getRange(1, 1, 1, headers[name].length).setValues([headers[name]]); sheet.getRange(1, 1, 1, headers[name].length).setFontWeight('bold'); sheet.setFrozenRows(1); } } return sheet; }
3
Deploy as Web App
In Apps Script, click Deploy → New Deployment. Set type to Web App. Set "Execute as" to Me. Set "Who has access" to Anyone. Click Deploy. Copy the deployment URL — it looks like https://script.google.com/macros/s/ABC123.../exec
⚠️ Important: After deploying, Google will ask you to authorize the script. Click "Advanced" → "Go to CompTIA A+ (unsafe)" → Allow. This is normal for Google Apps Scripts.
4
Paste URL above and test
Paste your deployment URL into the field at the top of this page and click Connect. All six platform tools will automatically sync to your Google Sheet. Student data will now follow their code across any device.
✓ Once connected: Every quiz score, PBIS point, note save, and question submission automatically writes to your Google Sheet in real time. You can view the sheet anytime for a complete record.
5
Share the Sheet URL with admin (optional)
Your Google Sheet is a live gradebook. You can share it (view-only) with your department head or admin directly from Google Drive. They'll see all student codes, scores, and PBIS points update in real time — no platform login needed.

⚠️ Rules & Reminders
→Your student code is STU-00X — never use your real name
→Timed practice IS graded — cold retrieval is NOT graded
→Lockdown quiz: do not switch tabs, right-click, or copy — violations are logged
→Earn PBIS points: 80%+ on quiz (+5), 7-day streak (+10), study session (+3)
→Confusion Flag is anonymous — use it! Your teacher wants to know
→CompTIA A+ passing score: 700 out of 900 points (~78%)
→Qualifier words on every question: BEST · FIRST · MOST · NEXT · LEAST
CompTIA Thinking Process — Use on Every Question
1.Find the qualifier word (BEST, FIRST, MOST, NEXT)
2.Pull evidence from the scenario or question stem
3.Eliminate two obviously wrong answers first
4.Choose the BEST answer — not just a correct one
