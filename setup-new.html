Apps Script (copy entire script)
function doGet(e){return handleRequest(e);} function doPost(e){return handleRequest(e);} function handleRequest(e){ const p=e.parameter; const ss=SpreadsheetApp.getActiveSpreadsheet(); try{ let result={}; if(p.action==='ping'){result={ok:true,message:'Connected!'};} else if(p.action==='save_login'){getOrCreate(ss,'Logins').appendRow([new Date(),p.code,p.tool,p.date,p.time]);result={ok:true};} else if(p.action==='save_score'){getOrCreate(ss,'Scores').appendRow([new Date(),p.code,p.obj,p.type,p.score,p.pct]);result={ok:true};} else if(p.action==='save_grade'){ const s=getOrCreate(ss,'Grades'); const data=s.getDataRange().getValues(); const i=data.findIndex(r=>r[1]===p.code); const row=[new Date(),p.code,Number(p.notes_pct||0),Number(p.disc_pct||0),Number(p.test_pct||0),Number(p.final_pct||0),p.letter||'']; if(i>0)s.getRange(i+1,1,1,row.length).setValues([row]); else s.appendRow(row); result={ok:true}; } else if(p.action==='get_class_summary'){ const scores=getOrCreate(ss,'Scores').getDataRange().getValues().slice(1); const codes=[...new Set(scores.map(r=>r[1]).filter(Boolean))]; const summary=codes.map(code=>{ const s=scores.filter(r=>r[1]===code); const t=s.filter(r=>r[3]==='quiz').map(r=>Number(r[5])); const avg=t.length?Math.round(t.reduce((a,b)=>a+b,0)/t.length):0; return{code,avg,sessions:s.length}; }); result={ok:true,summary}; } return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); }catch(err){return ContentService.createTextOutput(JSON.stringify({ok:false,error:err.message})).setMimeType(ContentService.MimeType.JSON);} } function getOrCreate(ss,name){ let s=ss.getSheetByName(name); if(!s){ s=ss.insertSheet(name); const h={Scores:['Timestamp','Code','Objective','Type','Score','Pct'],Logins:['Timestamp','Code','Tool','Date','Time'],Grades:['Timestamp','Code','Notes+Quizzes %','Discussion %','Tests %','Final Grade %','Letter']}; if(h[name]){s.getRange(1,1,1,h[name].length).setValues([h[name]]).setFontWeight('bold');s.setFrozenRows(1);} } return s; }
